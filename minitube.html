package main import ( "fmt" "io" "log" "net/http" "os" "path/filepath" "sync" "time" "github.com/gin-gonic/gin" ) const ( uploadDir = "./uploads" maxFileSize = 500 << 20 // 500 MB ) var ( mu sync.Mutex uploadedFiles = make(map[string]string) // filename → original name ) func main() { if err := os.MkdirAll(uploadDir, 0755); err != nil { log.Fatal(err) } r := gin.Default() r.Use(func(c *gin.Context) { c.Writer.Header().Set("Access-Control-Allow-Origin", "*") c.Writer.Header().Set("Access-Control-Allow-Methods", "GET, POST, OPTIONS") c.Writer.Header().Set("Access-Control-Allow-Headers", "*") if c.Request.Method == "OPTIONS" { c.AbortWithStatus(204) return } c.Next() }) r.GET("/", func(c *gin.Context) { c.String(http.StatusOK, "POST /upload → upload video (multipart field: video)\nGET /videos → list uploaded files\nGET /play/:name → stream video") }) r.POST("/upload", handleUpload) r.GET("/videos", func(c *gin.Context) { mu.Lock() defer mu.Unlock() files := make([]string, 0, len(uploadedFiles)) for name := range uploadedFiles { files = append(files, name) } c.JSON(http.StatusOK, gin.H{ "files": files, "count": len(files), }) }) r.GET("/play/:filename", func(c *gin.Context) { filename := c.Param("filename") path := filepath.Join(uploadDir, filename) if _, err := os.Stat(path); os.IsNotExist(err) { c.JSON(http.StatusNotFound, gin.H{"error": "file not found"}) return } c.Header("Content-Type", "video/mp4") c.Header("Accept-Ranges", "bytes") c.File(path) }) log.Println("server running on :8080") log.Fatal(r.Run(":8080")) } func handleUpload(c *gin.Context) { c.Request.Body = http.MaxBytesReader(c.Writer, c.Request.Body, maxFileSize) file, header, err := c.Request.FormFile("video") if err != nil { c.JSON(http.StatusBadRequest, gin.H{"error": "missing or invalid video field"}) return } defer file.Close() if header.Size > maxFileSize { c.JSON(http.StatusRequestEntityTooLarge, gin.H{"error": "file too large (max 500 MB)"}) return } ext := filepath.Ext(header.Filename) if ext != ".mp4" && ext != ".webm" { c.JSON(http.StatusBadRequest, gin.H{"error": "only .mp4 and .webm allowed"}) return } timestamp := time.Now().Format("20060102-150405") safeName := fmt.Sprintf("%s%s", timestamp, ext) fullPath := filepath.Join(uploadDir, safeName) out, err := os.Create(fullPath) if err != nil { c.JSON(http.StatusInternalServerError, gin.H{"error": "cannot create file"}) return } defer out.Close() _, err = io.Copy(out, file) if err != nil { c.JSON(http.StatusInternalServerError, gin.H{"error": "cannot save file"}) return } mu.Lock() uploadedFiles[safeName] = header.Filename mu.Unlock() c.JSON(http.StatusOK, gin.H{ "message": "upload successful", "filename": safeName, "url": fmt.Sprintf("http://localhost:8080/play/%s", safeName), }) }
